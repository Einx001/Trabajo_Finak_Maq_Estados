void read_temperature(void);

//Estados finitos de la maquina
enum estado{
  estado1Seguridad,// sistema bloqueado
  estado2Configuracion,// estado de configuracion
  estado3Monitoreo,//Donde suceden los hechos
  estado4Alarma// estado de alarma
};
estado currentState;//controlador de estados

enum Input{
  Unknown,
  reset,
  forward,
  Backward
};
Input currentInput;//controlador de entradas

//procesador de la funcion entregada por el fotoresistor
void read_photoresistor(){
  photo_value = analogRead(pin_photo);
}
//Procesador de la informacion entregada por el sensor de temperatura
void read_temperature(){
  //Extraer temperatura en grados centigrados
  long a = 1023 - analogRead(pin_temp);
  temperatura = beta / (log((1025.0 * 10 / a - 10) / 10) + beta / 298.0) - 273.0;
}

void setup(){
  //Inicializacion de pines
  //LED RGB
  pinMode(LED_RED, OUTPUT);
  pinMode(LED_GREEN, OUTPUT);
  pinMode(LED_BLUE, OUTPUT);
  //Buzzer
  pinMode(pin_buzz, OUTPUT);
  Serial.begin(9600);
  //Inicializacion de la pantalla LCD
  lcd.begin(16, 2);
  lcd.clear();
  //Inicializacion de las tareas asincronicas
  asyncTaskTemp.Start();
  asyncTaskPhoto.Start();
  //Inicializacion de la maquina de estados
  currentState =  estado::estado1Seguridad;
  currentInput = Input::Unknown;
}

void loop(){
 char customKey = keypad.getKey();
  if (customKey){
    press_key_static(customKey);
  }
  updateStateMachine();
}

//Funcion que administra los estados de la maquina de estados
void updateStateMachine(){
  switch (currentState){
    case estado::estado1Seguridad:
        switch (currentInput){
        case Input::Unknown: Seguridad(); break;
        case Input::reset:
            currentState = estado::estado1Seguridad; 
            auxSeguridad = true;
            currentInput = Input::Unknown;
            lcd.clear();
            lcd.setCursor(0, 0);
           break;
        case Input::forward:
            currentState = estado::estado2Configuracion; 
            auxSeguridad  = true;
            currentInput = Input::Unknown;
            lcd.clear();
            lcd.setCursor(0, 0);
          break;
        case Input::Backward:
            currentState = estado::estado1Seguridad; 
            auxSeguridad  = true;
            currentInput = Input::Unknown;
            lcd.clear();
            lcd.setCursor(0, 0);
           break;
        default: break;
      }
      break;
    case estado::estado2Configuracion:
        switch (currentInput){
          case Input::Unknown: Configuracion(); break;
          case Input::reset:
              currentState = estado::estado1Seguridad; 
              auxSeguridad  = true;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::forward:
              currentState = estado::estado3Monitoreo; 
              auxSeguridad  = true;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::Backward:
              currentState = estado::estado2Configuracion; 
              auxSeguridad  = true;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          default: break;
        }
       break;
    case estado::estado3Monitoreo:
     switch (currentInput){
          case Input::Unknown: funcion_monitoreo(); break;
          case Input::reset: 
              currentState = estado::estado1Seguridad; 
              auxMonitoreo = true;
              ContrasenaCorrecta = false;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::forward:
              currentState = estado::estado4Alarma; 
              auxMonitoreo = true;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::Backward:
              currentState = estado::estado2Configuracion; 
              auxMonitoreo = true;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          default: break;
        }
      break;
    case estado::estado4Alarma:
     switch (currentInput){
          case Input::Unknown: Alarma(); break;
          case Input::reset:
              currentState = estado::estado1Seguridad;
              ContrasenaCorrecta = false;
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::forward:
              currentState = estado::estado3Monitoreo; 
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          case Input::Backward:
              currentState = estado::estado3Monitoreo; 
              currentInput = Input::Unknown;
              lcd.clear();
              lcd.setCursor(0, 0);
            break;
          default: break;
        }
      break;
    default: break;
  }
}

void press_key_static(char key){ // este metodo me ejecuta la lectura de la entrada desde las teclas 8 (back) y 9 (reset)
  if (key == 'A') {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("BACK");
    Serial.println("BACK");
    delay(500);
    currentInput = Input::Backward;
  } 
  if (key == 'B') {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("RESET");
    Serial.println("RESET");
    delay(500);
    currentInput = Input::reset;
  }
}

//Estado de Seguridad
void Seguridad() {
  if (auxSeguridad){
    indice = 0;
    intentos = 0;
    cursor = 0;
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Ingrese Clave:");
    lcd.setCursor(0, 1);
    auxSeguridad = false;
  }
  char customKey = keypad.getKey();
  if (customKey){
    lcd.print(customKey);
    cursor++;
    if (customKey == contrasena[indice] && cursor <= 4) {
      indice ++;
      if (indice == 4) {
        lcd.clear();
        lcd.setCursor(0, 0); 
        lcd.print("Clave Correcta!");
        digitalWrite(LED_GREEN, HIGH);
        delay(3000);
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Bienvenido al");
        lcd.setCursor(0, 1);
        lcd.print("Sistema!");
        delay(3000);
        digitalWrite(LED_GREEN, LOW);
        ContrasenaCorrecta = true;
        indice = 0;
        intentos = 0;
        cursor = 0;
        lcd.clear();
        lcd.setCursor(0, 0);
        delay(1000);
      }
    } else {
      indice = 0;
    }
    if (cursor >= 4) {
      cursor = 0;
      indice = 0;
      intentos++;
      lcd.setCursor(0, 1);
      lcd.print("Clave Incorrecta");
      digitalWrite(LED_BLUE, HIGH);
      delay(2000);
      digitalWrite(LED_BLUE, LOW);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Ingrese Clave");
      lcd.setCursor(0, 1);
      //Sistema bloqueado
      if (intentos == 3) {
        intentos = 0;
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("El Sistema se ha");
        lcd.setCursor(0, 1);
        lcd.print("Bloqueado!");
        //led rojo
        digitalWrite(LED_RED, HIGH);
        delay(3000);
        digitalWrite(LED_RED, LOW);
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Ingrese la clave");
        lcd.setCursor(0, 1);
      }
    }
  }
  if (ContrasenaCorrecta) {
    currentInput = Input::forward;
    ContrasenaCorrecta = false;
    lcd.clear();
    lcd.setCursor(0, 0);
  }
}

//Variables para obtener valores a partir del teclado y definir umbrales
char valIngresado[3];
int indiceVal=0;
char auxString;
int numero;

void OpcionUmbTempHigh(){
  posMemory = 0;
  byte tempMemory = EEPROM.read(posMemory);
  if (tempMemory != NULL) {
    umbTempHigh = tempMemory;
  }
  bool tempFlag = true;
  lcd.setCursor(0, 1);
  while(tempFlag) {
    char customKey = keypad.getKey();
    if (customKey){
      valIngresado[indiceVal]=customKey;
      lcd.print(customKey);
      indiceVal++;
      if (indiceVal == 2){
        defUmbTempHigh = atoi(valIngresado);
        tempFlag = false;
        indiceVal = 0;
        EEPROM.put(posMemory, defUmbTempHigh);
      }
    } 
  }
}

void OpcionReset(){
  posMemory = 0;
  EEPROM.update(posMemory, defUmbTempHigh);
  posMemory = 1;
  EEPROM.update(posMemory, defUmbTempLow);
  posMemory = 2;
  EEPROM.update(posMemory, defUmbLuz);
  posMemory = 3;
  EEPROM.update(posMemory, defUmbLuzLow);
  lcd.setCursor(0, 1);
  lcd.print("Umbrales por");
  lcd.setCursor(0, 1);
  lcd.print("Defecto");
  delay(1000);
  auxConfiguracion = true;
  lcd.clear();
  lcd.setCursor(0, 0);
}

//Estado de monitoreo
void start_monitoreo(){
  leer_Eprom();
}

void funcion_monitoreo() {
  if (auxMonitoreo) { 
    start_monitoreo();
    auxMonitoreo = false;
  }
  // se actualizan las tareas asincronas
  asyncTaskTemp.Update();
  asyncTaskPhoto.Update();
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Temp:");
  lcd.print(temperatura);
  lcd.setCursor(0, 1);
  lcd.print("Luz:");
  lcd.print(photo_value);
  delay(1000);
  // seleccion de funciones segun temperatura
  if (temperatura >= umbTempHigh) {
    currentInput = Input::forward;
  }
  if (temperatura <= umbTempLow) {
    currentInput = Input::forward;
  }
  if (photo_value >= umbLuzHigh) {
    currentInput = Input::forward;
  }
  if (photo_value <= umbLuzLow) {
    currentInput = Input::forward;
  }
}

//Lee los valores de los umbrales almacenados en la memoria
void leer_Eprom() {
  posMemory = 0;
  int auxMemory = EEPROM.read(posMemory);
  if (auxMemory != NULL) {
    umbTempHigh = auxMemory;
  }
  posMemory = 1;
  auxMemory = EEPROM.read(posMemory);
  if (auxMemory != NULL) {
    umbTempLow = auxMemory;
  }
  posMemory = 2;
  auxMemory = EEPROM.read(posMemory);
  if (auxMemory != NULL) {
    umbLuzHigh = auxMemory;
  }
  posMemory = 3;
  EEPROM.get(posMemory, auxMemory);
  if (auxMemory != NULL) {
    umbLuzLow = auxMemory;
  }
}

//Estado de alarma
void Alarma() {
  digitalWrite(LED_RED, HIGH);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Alarma Umbral");
  lcd.setCursor(0, 1);
  lcd.print("Superado");
  sonar_buzz();
  delay(3000);
  digitalWrite(LED_RED, LOW);
  tiempo = millis(); // se da tiempo de espera de alarma
  while (millis() < tiempo + 2000UL) {
    currentInput = Input::forward;
  }
}

//Funcion encargada de hacer sonar el buzzer
void sonar_buzz() {
  for (int i = 0; i < 3; i++) {
    int duracion = 1000 / duraciones[i];
    tone(pin_buzz, melodia[i], duracion);
    int pausa = duracion * 1.30;
    delay(pausa);
    noTone(pin_buzz);
  }
}
